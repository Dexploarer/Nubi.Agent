### Option 2: Manual Installation

<Tabs>
  <TabItem label="Linux/macOS">
```bash
# 1. Install Node.js (using Node Version Manager)
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.0/install.sh | bash
source ~/.bashrc
nvm install 20
nvm use 20

# 2. Install Bun (recommended package manager)

curl -fsSL https://bun.sh/install | bash
source ~/.bashrc

# 3. Clone NUBI repository

git clone https://github.com/anubis-chat/nubi.git
cd nubi

# 4. Install dependencies

bun install

# 5. Verify installation

bun run type-check
bun test

````
  </TabItem>

  <TabItem label="Windows (WSL2)">

### Production Deployment Options

<Tabs>
  <TabItem label="Traditional Server">
```bash
# 1. Server preparation (Ubuntu 22.04 LTS)
sudo apt update && sudo apt upgrade -y
sudo apt install -y nodejs npm postgresql-14 nginx redis-server

# 2. User and directory setup
sudo useradd -m -s /bin/bash nubi
sudo mkdir -p /opt/nubi
sudo chown nubi:nubi /opt/nubi

# 3. Switch to nubi user
sudo su - nubi

# 4. Install production dependencies
curl -fsSL https://bun.sh/install | bash
source ~/.bashrc

# 5. Clone and build
cd /opt/nubi
git clone https://github.com/anubis-chat/nubi.git .
bun install --production
bun run build

# 6. Environment configuration
cp .env.production.example .env.production
# Edit .env.production with production values

# 7. Database setup
createdb nubi_production
bun run db:migrate

# 8. Process manager setup
npm install -g pm2
pm2 start ecosystem.config.js --env production
pm2 startup
pm2 save
````

  </TabItem>
  
  <TabItem label="Docker Production">
```bash
# 1. Multi-stage production build
cat > Dockerfile.prod << 'EOF'
FROM oven/bun:1-slim as base
WORKDIR /app

# Install dependencies

FROM base as deps
COPY package.json bun.lockb ./
RUN bun install --frozen-lockfile --production

# Build stage

FROM base as build  
COPY . .
COPY --from=deps /app/node_modules ./node_modules
RUN bun run build

# Production runtime

FROM base as runtime
COPY --from=deps /app/node_modules ./node_modules
COPY --from=build /app/dist ./dist
COPY package.json ./

EXPOSE 3001
CMD ["bun", "start"]
EOF

# 2. Build production image

docker build -f Dockerfile.prod -t nubi:prod .

# 3. Production compose stack

cat > docker-compose.prod.yml << 'EOF'
version: '3.8'
services:
nubi:
image: nubi:prod
restart: unless-stopped
ports: - "3001:3001"
environment: - NODE_ENV=production - POSTGRES_URL=postgresql://user:pass@postgres:5432/nubi
depends_on: - postgres - redis

postgres:
image: postgres:15-alpine
restart: unless-stopped
volumes: - postgres_data:/var/lib/postgresql/data
environment:
POSTGRES_PASSWORD: ${POSTGRES_PASSWORD}

redis:
image: redis:7-alpine
restart: unless-stopped
volumes: - redis_data:/data

nginx:
image: nginx:alpine
restart: unless-stopped
ports: - "80:80" - "443:443"
volumes: - ./nginx.conf:/etc/nginx/nginx.conf - ./ssl:/etc/nginx/ssl

volumes:
postgres_data:
redis_data:
EOF

# 4. Deploy with compose

docker-compose -f docker-compose.prod.yml up -d

````

## üîê Security Hardening

### Essential Security Configuration

<Tabs>
  <TabItem label="Environment Security">
```bash
# 1. Secure environment variables
sudo mkdir -p /etc/nubi
sudo chmod 700 /etc/nubi

# 2. Create secure .env file

sudo cat > /etc/nubi/.env << 'EOF'
NODE_ENV=production
JWT_SECRET=$(openssl rand -base64 32)
ENCRYPTION_KEY=$(openssl rand -base64 32)
DB_ENCRYPTION_KEY=$(openssl rand -base64 32)

# AI Provider Keys (encrypted at rest)

OPENAI_API_KEY=sk-encrypted-key
ANTHROPIC_API_KEY=sk-encrypted-key

# Database with SSL

POSTGRES_URL=postgresql://user:pass@localhost:5432/nubi?sslmode=require

# Redis with AUTH

REDIS_URL=redis://user:pass@localhost:6379

# Telegram with webhook validation

TELEGRAM_BOT_TOKEN=encrypted-token
TELEGRAM_WEBHOOK_SECRET=$(openssl rand -base64 16)
EOF

sudo chown nubi:nubi /etc/nubi/.env
sudo chmod 600 /etc/nubi/.env

````
  </TabItem>

  <TabItem label="Network Security">
```bash
# 1. Configure firewall (UFW)
sudo ufw default deny incoming
sudo ufw default allow outgoing
sudo ufw allow ssh
sudo ufw allow 80/tcp
sudo ufw allow 443/tcp
sudo ufw allow from 10.0.0.0/8 to any port 5432  # Database access
sudo ufw enable

# 2. Nginx security configuration
sudo cat > /etc/nginx/sites-available/nubi << 'EOF'
server {
    listen 443 ssl http2;
    server_name your-domain.com;

    ssl_certificate /path/to/cert.pem;
    ssl_certificate_key /path/to/key.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-RSA-AES256-GCM-SHA512:DHE-RSA-AES256-GCM-SHA512;

    # Security headers
    add_header X-Frame-Options DENY;
    add_header X-Content-Type-Options nosniff;
    add_header X-XSS-Protection "1; mode=block";
    add_header Strict-Transport-Security "max-age=63072000";

    location / {
        proxy_pass http://localhost:3001;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # Rate limiting
        limit_req zone=api burst=20 nodelay;
    }
}
EOF

sudo ln -s /etc/nginx/sites-available/nubi /etc/nginx/sites-enabled/
sudo nginx -t && sudo systemctl reload nginx
````

  </TabItem>
  
  <TabItem label="Database Security">
```sql
-- PostgreSQL security configuration

-- 1. Create dedicated user
CREATE USER nubi_app WITH PASSWORD 'secure-random-password';

-- 2. Create database with proper ownership
CREATE DATABASE nubi_production OWNER nubi_app;

-- 3. Grant minimal required privileges
GRANT CONNECT ON DATABASE nubi_production TO nubi_app;
GRANT CREATE, USAGE ON SCHEMA public TO nubi_app;
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA public TO nubi_app;

-- 4. Enable Row Level Security
ALTER DATABASE nubi_production SET row_security = on;

-- 5. Configure connection limits
ALTER USER nubi_app CONNECTION LIMIT 20;

-- 6. SSL-only connections
ALTER SYSTEM SET ssl = on;
ALTER SYSTEM SET ssl_cert_file = '/path/to/server.crt';
ALTER SYSTEM SET ssl_key_file = '/path/to/server.key';

````
